Contactos = []
f = open('agenda.csv', 'r')

for linea in f:
    lista_datos = linea.split('|')
    lista_datos[2] = lista_datos[2].replace("\n", "")
    Contactos.append(lista_datos)

f.close()
print(Entradas)
print(Contactos)

import json
datos_json=json.dumps(Contactos,indent=4)
